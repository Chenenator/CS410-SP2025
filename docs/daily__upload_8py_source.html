38def get_max_min_temperature_from_today():
39 today = datetime.now().strftime("%d-%m-%Y")
40 today_csv_path = os.path.join(localWeatherFolder, f"{today}.csv")
41
42 if not os.path.exists(today_csv_path):
43 print(f"No data file for today: {today_csv_path}")
44 return None, None
45
46 try:
47 df = pd.read_csv(today_csv_path, encoding_errors='ignore')
48 df['Temp (F)'] = pd.to_numeric(df['Temp (F)'], errors='coerce')
49 max_temp = df['Temp (F)'].max()
50 min_temp = df['Temp (F)'].min()
51 return max_temp, min_temp
