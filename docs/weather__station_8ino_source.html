ESP32 IoT Weather Station with AI Forecasting 0.0.1
This is a smart, connected weather station powered by an ESP32 board and various sensors. It collects real-time environmental data, sends it to a dashboard, and uses AI to predict future weather trends based on historical data.

224void loop() {
225 // wait a 1 seconds between readings
226 delay(1000);
227
228 if (!reconnect()) return;
229 if (!tb.connected()) {
230 connectToThingsBoard();
231 }
232 // read humidity
233 float humi = dht11.readHumidity();
234
235 // Read light sensor
236 int lightValue = analogRead(LIGHT_SENSOR_PIN);
237 String brightness = "";
238 // We'll have a few threshholds, qualitatively determined
239 if (lightValue < 40) {
240 brightness = "Dark";
241 } else if (lightValue < 800) {
242 brightness = "Dim";
243 } else if (lightValue < 2000) {
244 brightness = "Light";
245 } else if (lightValue < 3200) {
246 brightness = "Bright";
247 } else {
248 brightness = "Very Bright";
249 }
250
251 // read the ADC value from the temperature sensor
252 long adcVal = analogRead(PIN_LM35);
253 // convert the ADC value to voltage in millivolt
254 float milliVolt = adcVal * (ADC_VREF_mV / ADC_RESOLUTION);
255 // convert the voltage to the temperature in °C
256 float tempC = milliVolt / 10;
257 // convert the °C to °F
258 float tempF = tempC * 9 / 5 + 32;
259
260 // check whether the reading is successful or not
261 if (isnan(humi) || isnan(adcVal)) {
262 Serial.println("Failed to read from DHT11 or LM35 sensor!");
263 } else {
264 // Print as CSV: humi,tempC,tempF
265 Serial.print(humi, 1);
266 Serial.print(",");
267 Serial.print(tempC, 1);
268 Serial.print(",");
269 Serial.print(tempF, 1);
270 Serial.print(",");
271 Serial.print(lightValue, 1);
272 Serial.print(",");
273 Serial.println(brightness);
274 sendDataToThingsBoard(tempC, tempF, humi, lightValue, brightness);
275 }
276
277 tb.loop();
278}
