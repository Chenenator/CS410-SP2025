49def list_ssid():
50 os_name = platform.system()
51 ssids = []
52
53 try:
54 if os_name == "Windows":
55 output = subprocess.check_output('netsh wlan show networks', shell=True, text=True)
56 # Extract SSIDs using regex (skip "SSID N : " format)
57 ssids = re.findall(r'\s+SSID\s+\d+\s+:\s(.+)', output)
58
59 elif os_name == "Linux":
60 output = subprocess.check_output("nmcli device wifi list", shell=True, text=True)
61 # Extract SSID names from each line (ignoring the header)
62 lines = output.strip().split('\n')[1:]
63 for line in lines:
64 parts = line.strip().split()
65 if parts:
66 ssids.append(parts[0])
67
68 else:
69 print("Unsupported OS")
70 except subprocess.CalledProcessError as e:
71 print(f"Error listing networks: {e}")
72
73 return ssids

80def get_current_network_credentials(ssid):
81 try:
82 # Get current connected SSID
83 current_output = subprocess.check_output(['netsh', 'wlan', 'show', 'interfaces'], encoding='utf-8')
84 ssid_match = re.search(r'\s*SSID\s*:\s(.+)', current_output)
85 if not ssid_match:
86 print("Could not find SSID.")
87 return
88
89 current_ssid = ssid_match.group(1).strip()
90 print(f"Current SSID: {current_ssid}")
91
92 # If the parameter is not empty, indicating that the current ssid could not be found,
93 # find the password of the manually inputted ssid.
94 if ssid is not None:
95 # Show profile details with key=clear
96 profile_output = subprocess.check_output(['netsh', 'wlan', 'show', 'profile', ssid, 'key=clear'],
97 encoding='utf-8')
98 else:
99 # Show profile details with key=clear
100 profile_output = subprocess.check_output(['netsh', 'wlan', 'show', 'profile', current_ssid, 'key=clear'],
101 encoding='utf-8')
102
103 # Obtain password of matching ssid.
104 key_match = re.search(r'Key Content\s*:\s(.+)', profile_output)
105
106 if key_match:
107 print(f"Password: {key_match.group(1).strip()}")
108 return current_ssid, key_match.group(1).strip()
109 else:
110 print("Password not found. Either on Eduroam or Open Network.")
111 return current_ssid, None
112
113 except subprocess.CalledProcessError as e:
114 print(f"Command failed: {e}")

121def update_wifi_credentials_in_ino(ssid, password):
122 # Creates a variable for the ino files and its path.
123 ino_file = [f for f in os.listdir(esp32_sketch_path) if f.endswith(".ino")][0]

217def gather_local_weather():
218 date_str = datetime.now().strftime("%d-%m-%Y")
219 os.makedirs(localWeatherFolder, exist_ok=True)
220 filename = os.path.join(localWeatherFolder, f"{date_str}.csv")
221
222 ser = serial.Serial(esp32_port, 9600, timeout=1)
223 file_exists = os.path.isfile(filename)
224
225 with open(filename, mode='a', newline='') as csv_file:
226 writer = csv.writer(csv_file)
227 if not file_exists:
228 writer.writerow(['Timestamp', 'Humidity (%)', 'Temp (C)', 'Temp (F)', 'Light Level (eV)', 'Brightness'])
229
230 print("📡 Listening to ESP32 serial output... Press Ctrl+C to stop.")
231 try:
232 while True:
233 line = ser.readline().decode('utf-8', errors='ignore').strip()
234 if line:
235 print(f"ESP32: {line}")
236 parts = line.split(',')
237 if len(parts) == 5:
238 timestamp = datetime.now().isoformat(timespec='seconds')
239 writer.writerow([timestamp, *[p.strip() for p in parts]])
240 csv_file.flush()
241
242 except KeyboardInterrupt:
243 print(" Data logging stopped.")
244 finally:
245 ser.close()
