35 def load_and_prepare_data(self):
36 # Load the weather data CSV into DataFrame and clean it
37 self.weather_data = pd.read_csv(self.data_path, index_col="DATE")
38 self.weather_data.columns = self.weather_data.columns.str.lower() # Optional
39
40 #Convert index to data type
41 self.weather_data.index = pd.to_datetime(self.weather_data.index, errors="coerce")
42 self.weather_data = self.weather_data.ffill() # Fill missing data
43
44 # Remove columns with more than 5% missing values
45 null_percentage = self.weather_data.apply(pd.isnull).sum() / self.weather_data.shape[0]
46 valid_columns = self.weather_data.columns[null_percentage < 0.05]
47 self.weather_data = self.weather_data[valid_columns].copy()
48
49 # Create target columns for prediction (next day's max and min temperature)
50 self.weather_data[["maxtemp", "mintemp"]] = self.weather_data[["tmax", "tmin"]].shift(-1)
51 self.weather_data = self.weather_data.drop(columns=["station", "name"], errors="ignore")
52
53 # Add rolling average feature
54 for horizon in [3, 14]:
55 for col in ["tmax", "tmin", "prcp"]:
56 self.compute_rolling(horizon, col)
57
58 self.weather_data = self.weather_data.iloc[14:, :] # Drop initial rows with NaNs
59 self.weather_data.fillna(0, inplace=True)
60
61 # Add expanding (cumulative mean) averages by month and day of year
62 for col in ["tmax", "tmin", "prcp"]:
63 self.weather_data[f"month_avg_{col}"] = self.weather_data[col].groupby(
64 self.weather_data.index.month, group_keys=False).apply(lambda x: x.expanding(1).mean())
65 self.weather_data[f"day_avg_{col}"] = self.weather_data[col].groupby(
66 self.weather_data.index.day_of_year, group_keys=False).apply(lambda x: x.expanding(1).mean())
67
68 # Final clean up
69 self.weather_data.replace([np.inf, -np.inf], np.nan, inplace=True)
70 self.weather_data.fillna(0, inplace=True)

76 def compute_rolling(self, horizon, col):
77 # Compute rolling average and percentage difference for the given column
78 label = f"rolling_{horizon}_{col}"
79 self.weather_data[label] = self.weather_data[col].rolling(horizon).mean()
80 self.weather_data[f"{label}_percentage"] = (self.weather_data[label] - self.weather_data[col]) / \
81 self.weather_data[col]

97 def predict_next_days(self, days):
98 # Predict the next N days of max and min temperature
99 predictors = self.get_predictors()
100 last_known = self.weather_data.copy()
101 predictions = []
102
103 for _ in range(days):
104 input_row = last_known.iloc[-1:][predictors] # Use last known data for prediction
105
106 self.model.fit(last_known[predictors], last_known["maxtemp"])
107 predicted_max = self.model.predict(input_row)[0] # predict max temp
108
109 self.model.fit(last_known[predictors], last_known["mintemp"])
110 predicted_min = self.model.predict(input_row)[0] # predict min temp
111
112 prediction_date = last_known.index[-1] + pd.Timedelta(days=1)
113
114 # Save prediction in a dictionary
115 predictions.append({
116 "date": prediction_date.strftime("%Y-%m-%d"),
117 "maxTemp": predicted_max,
118 "minTemp": predicted_min
119 })
120 # Simulate new row for the next prediction iteration
121 new_row = last_known.iloc[-1:].copy()
122 new_row.index = [prediction_date]
123 new_row["tmax"] = predicted_max
124 new_row["tmin"] = predicted_min
125 new_row["maxtemp"] = predicted_max
126 new_row["mintemp"] = predicted_min
127 last_known = pd.concat([last_known, new_row])
128
129 return predictions

165def send_to_thingsboard(forecast, token):
166 url = f"https://thingsboard.cloud/api/v1/{token}/telemetry"
167
168 for day in forecast:
169 # Convert forecast date to timestamp in milliseconds
170 timestamp = int(datetime.strptime(day["date"], "%Y-%m-%d").timestamp() * 1000)
171
172 payload = {
173 "ts": timestamp,
174 "values": {
175 "Predicted maxTemp": round(day["maxTemp"], 2),
176 "Predicted minTemp": round(day["minTemp"], 2)
177 }
178 }
179
180 # Send each day's payload separately
181 response = requests.post(url, json=payload)
182
183 if response.status_code != 200:
184 print(f"Error sending data for {day['date']}: {response.status_code} {response.text}")
185 else:
186 print(f"Sent prediction for {day['date']}")
