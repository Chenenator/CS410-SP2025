47* @brief Function gets brightness value

50* @returns String the string representation of brightness levels
51*/
52async function getBrightness() {
53 // Check if token is expired before making request
54 if (!jwtToken || Date.now() >= tokenExpires) {
55 await loginToThingsboard(); // Refresh token: login and get token
56 }
57
58 const res = await axios.get(
59 `${TB_BASE_URL}/api/plugins/telemetry/DEVICE/${DEVICE_ID}/values/timeseries?keys=Brightness`,
60 {
61 headers: { Authorization: `Bearer ${jwtToken}` },

65 // Parse brightness from response data using optional chaining to avoid null errors

67 console.log("Brightness:", brightness);
